# single line comment
```
#include <iostream> <--- this line will  be executed
int main (){        <--- this line will  be executed
                    <--- this line will  be executed { considered as whitespace}
    // hey ace      <--- this line will  be skipped               
    return 0;       <--- this line will  be executed
}                   <--- this line will  be executed
```

# multiline comment
```
#include <iostream>
int main (){        
                    
    /* example of multiline */               
    return 0;
} 
```
